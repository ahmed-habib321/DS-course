    private int size;
    private Object[] arr;
    private int top;

    public ArrB(int size) {
        this.size = size;
        arr = new Object[size];
        this.top = 0;
    }

    void push(Object data) {
        if (top < size) {
            arr[top] = data;
            top++;
        }
    }

    Object pop() {
        if (top > 0) {
            top--;
            return arr[top];
        } else {
            return null;
        }

    }

    boolean isEmpty() {
        return top == 0;
    }

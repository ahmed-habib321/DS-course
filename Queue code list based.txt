    private node head = null;
    private node tail = null;

    public void EnQueue(Object data) {
        node node = new node(data);
        if (tail == null) {
            head = tail = node;
        } else {
            tail.next = node;
            tail = node;
        }
    }

    public Object deQueue() {
        Object t = null;
        t = head.data;
        head = head.next;
        if (head == null) {
            head = tail = null;
        }
        return t;
    }

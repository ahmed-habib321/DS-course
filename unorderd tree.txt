child root = null;
   

    public boolean isEmpty() {
        return (root == null);
    }        

    
    int sum(child root){
        if (root == null) {
            return 0;
        }        
        return root.data + sum(root.right) + sum(root.left);    
    }
    
    int count(child root) {
        if (root == null) {
            return 0;
        }
        return 1 + count(root.right) + count(root.left);
    }

    int countleave(child root) {
        if (root == null) {
            return 0;
        }
        if (root.left == null && root.right == null) {
            return 1;
        } else {
            return 1 + countleave(root.right) + countleave(root.left);
        }
    }

    boolean isidentical(child root1, child root2) {
        if (root1 == null && root2 == null) {
            return true;
        }
        if (root1 == null || root2 == null) {
            return false;
        }
        if (root1.data == root2.data) {
            return true;
        }
        return isidentical(root1.right, root2.right) && isidentical(root1.left, root2.left);

    }
    
    public void Insert(int data) {
        root = unorderedInsertNode(root, data);
    }

    private child unorderedInsertNode(child node, int data) {
        if (node == null) {
            node = new child(data);
        } else if (node.right == null) {
            node.right = unorderedInsertNode(node.right, data);
        } else {
            node.left = unorderedInsertNode(node.left, data);
        }
        return node;
    }
